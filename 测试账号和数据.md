### 查看数据
```sql
-- 查看所有游戏
SELECT * FROM "Game";

-- 查看所有用户
SELECT "username", "role", "realName" FROM "User";

-- 查看所有问题类型
SELECT "name", "priorityWeight", "icon", "sortOrder"
FROM "IssueType"
ORDER BY "sortOrder";

-- 查看最新的工单
SELECT 
  t."ticketNo",
  g."name" as "游戏",
  t."playerIdOrName" as "玩家",
  t."priority",
  t."priorityScore",
  t."status",
  t."createdAt"
FROM "Ticket" t
JOIN "Game" g ON t."gameId" = g."id"
ORDER BY t."createdAt" DESC
LIMIT 10;
```
